// Converteste data in format '12-Oct-1984' function getDateString(dt) { return dt.getDate() + '-' + ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'][dt.getMonth()] + '-' + dt.getFullYear(); } // Converteste data in format 'July 2010' function getMonthYearString(dt) { return ['January','February','March','April','May','June','July', 'August','September','October','November','December'][dt.getMonth()] + ' ' + dt.getFullYear(); } // Functie apelata cand userul da click pe orice buton function chooseDate(e) { var targ; if (!e) var e = window.event; if (e.target) targ = e.target; else if (e.srcElement) targ = e.srcElement; if (targ.nodeType == 3) targ = targ.parentNode; // gaseste text-boxul if (targ.value=='<' || targ.value=='>') { createCalendar(div, new Date(targ.getAttribute('date'))); return; } textbox.value = targ.getAttribute('date'); // Seteaza data selectata div.parentNode.removeChild(div); } // Converteste data in format d-MMM-yyyy function parseMyDate(d) { if (d=="") return new Date('NotADate'); // PT safari var a = d.split('-'); if (a.length!=3) return new Date(d); var m = -1; if (a[1]=='Jan') m=0; if (a[1]=='Feb') m=1; if (a[1]=='Mar') m=2; if (a[1]=='Apr') m=3; if (a[1]=='May') m=4; if (a[1]=='Jun') m=5; if (a[1]=='Jul') m=6; if (a[1]=='Aug') m=7; if (a[1]=='Sep') m=8; if (a[1]=='Oct') m=9; if (a[1]=='Nov') m=10; if (a[1]=='Dec') m=11; if (m<0) return new Date(d); // Nu s-a gasit luna return new Date(a[2],m,a[0],0,0,0,0); } // Creaza calendarul pentru o luna data function createCalendar(div, month) { var idOfTextbox = div.getAttribute('datepickertextbox');
